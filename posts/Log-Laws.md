$$
Log_a(mn) = log_a(m) + log_a(n)
$$

$$
Log_a(m^p) = pLog_a(m)
$$

$$
Log_a({m \over n}) = log_a(m) - log_a(n)
$$

$$
Log_a(a) = 1
$$

$$
Log_a(1) = 0
$$

$$
Log_a({1\over n}) = -Log_a(n)
$$
